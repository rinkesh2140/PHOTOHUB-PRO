💾 DATABASE CHANGES NEEDED

Step 1: Add columns to existing tables

ALTER TABLE photos ADD COLUMN event_time_label VARCHAR(100);
ALTER TABLE photos ADD COLUMN event_time_order INT DEFAULT 0;

Step 2: Create new tables

CREATE TABLE slideshows (
  id SERIAL PRIMARY KEY,
  project_id INT,
  name VARCHAR(255),
  music_style VARCHAR(50), -- romantic, upbeat, emotional, classic
  created_at TIMESTAMP,
  FOREIGN KEY (project_id) REFERENCES projects(id)
);

CREATE TABLE slideshow_photos (
  id SERIAL PRIMARY KEY,
  slideshow_id INT,
  photo_id INT,
  order_number INT,
  FOREIGN KEY (slideshow_id) REFERENCES slideshows(id),
  FOREIGN KEY (photo_id) REFERENCES photos(id)
);

CREATE TABLE photo_tags (
  id SERIAL PRIMARY KEY,
  photo_id INT,
  person_name VARCHAR(255),
  face_index INT,
  created_at TIMESTAMP,
  FOREIGN KEY (photo_id) REFERENCES photos(id)
);

CREATE TABLE instagram_stories (
  id SERIAL PRIMARY KEY,
  project_id INT,
  photo_id INT,
  template_type VARCHAR(50),
  text_overlay VARCHAR(255),
  hashtags VARCHAR(500),
  created_at TIMESTAMP,
  FOREIGN KEY (project_id) REFERENCES projects(id)
);

CREATE TABLE anniversary_reminders (
  id SERIAL PRIMARY KEY,
  project_id INT,
  event_date DATE,
  reminder_sent_count INT DEFAULT 0,
  photoshoot_claimed BOOLEAN DEFAULT FALSE,
  photoshoot_date DATE,
  created_at TIMESTAMP,
  FOREIGN KEY (project_id) REFERENCES projects(id)
);

Step 3: Create indexes for performance

CREATE INDEX idx_photo_tags_person ON photo_tags(person_name);
CREATE INDEX idx_slideshow_photos_order ON slideshow_photos(order_number);
CREATE INDEX idx_anniversary_event_date ON anniversary_reminders(event_date);
